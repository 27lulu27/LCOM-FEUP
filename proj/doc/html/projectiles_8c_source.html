66
67 if (projectile->type == PLAYER_PROJECTILE) {
68 //check collision with aliens
69 for (int i = 0; i < ALIEN_COUNT; i++) {
70 if (aliens[i].alive) {
71 if (proj_x >= aliens[i].x && proj_x <= aliens[i].x + 60 &&
72 proj_y >= aliens[i].y && proj_y <= aliens[i].y + 45) {
73 *alien_index = i;
74 return true;
75 }
76 }
77 }
78 } else if (projectile->type == ALIEN_PROJECTILE) {
79 //check collision with player
80 if (proj_x >= player_x && proj_x <= player_x + 60 &&
81 proj_y >= player_y && proj_y <= player_y + 45) {

181int calc_x(Projectile* projectile) {
182 int x = projectile->xi + projectile->speed * cos(atan2((projectile->yf - projectile->yi), (projectile->xf - projectile->xi))) * projectile->t;

186int calc_y(Projectile* projectile) {
187 int y = projectile->yi + projectile->speed * sin(atan2((projectile->yf - projectile->yi), (projectile->xf - projectile->xi))) * projectile->t;
